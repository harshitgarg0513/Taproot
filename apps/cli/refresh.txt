import {
  Injectable,
  CanActivate,
  ExecutionContext,
} from '@nestjs/common';
import { JwtService } from '@nestjs/jwt';
import { UnAuthorizedException } from '../exceptions/unauthorized.exception';

@Injectable()
export class AuthGuard implements CanActivate {
  constructor(private readonly jwtService: JwtService) {}

  async canActivate(context: ExecutionContext): Promise<boolean> {
    const call = context.switchToRpc().getContext();
    const metadata = call.internalRepr;
    const jwtAuthorization = metadata.get('authorization');
    const jwtToken = jwtAuthorization ? jwtAuthorization[0] : null;

    if (!this.shouldSkipAuthorization(context)) {
      try {
        const decodedToken = this.jwtService.verify(jwtToken);
        call.identity = decodedToken;
      } catch (e) {
        call.identity = null;
        throw new UnAuthorizedException('Unauthorized');
      }
    }

    return true;
  }

  private shouldSkipAuthorization(context: ExecutionContext): boolean {
    const ctx = context.switchToRpc();

    const path = ctx['args'][2]?.['path'];

    const publicEndpoints = ['/beatroute.dms.HealthService/CheckHealth'];
    return publicEndpoints.includes(path);
  }
}

import { InterceptingCall } from '@grpc/grpc-js';
import { Inject, Injectable } from '@nestjs/common';
import { REQUEST } from '@nestjs/core';

@Injectable()
export class GrpcClientInterceptor {
  constructor(@Inject(REQUEST) private readonly request: any) {}

  intercept(options, nextCall) {
    const jwtToken = this.request['jwtToken'];
    return new InterceptingCall(nextCall(options), {
      start: (metadata, listener, next) => {
        metadata.add('authorization', `${jwtToken}`);
        next(metadata, listener);
      },
    });
  }
}

To implement Refresh Token Rotation, we need to ensure that every time a refresh token is used, it is invalidated and a new pair (Access + Refresh) is issued. If a client attempts to use an already-used or expired refresh token, we must flag this as a potential security breach (token reuse detection).

### 1. Database Schema Update
We need a way to track the "family" of refresh tokens to detect reuse. 

**File:** `apps/microservice/src/core/team/entities/user.entity.ts`
Add a column to store a `refresh_token_version` or a specific `refresh_token_hash`. For rotation, tracking the version is standard.

```typescript
// ... inside User class
  @Column({ default: 0 })
  refresh_token_version: number;
```
*Change explanation:* We add `refresh_token_version`. When a user logs in, we increment this. When a refresh token is presented, we check if the version in the token matches the DB. If it doesn't match, it means the token was already rotated/compromised.

### 2. Auth Service Logic (Logic Requirements)
Since I cannot see the `AuthService` implementation, you must update your `authenticate` logic in the microservice:

1.  **Issue Tokens:** When issuing a new token pair, include `sub` (user id) and `version` (the user's `refresh_token_version`).
2.  **Verify Refresh:**
    *   Decode the refresh token.
    *   Find the user in the DB.
    *   **Check:** `if (decoded.version !== user.refresh_token_version) throw new SecurityException('Token Reused');`
    *   If valid, increment `user.refresh_token_version` in the DB and return a new token pair.

### 3. Middleware Update (API Gateway)
The `RequestMiddleware` currently only handles `authenticate`. It needs to be aware of the "Refresh" flow.

**File:** `apps/apigateway/src/middlewares/request/request.middleware.ts`

The current `GrpcClientInterceptor` adds the token to metadata. This remains the same as the rotation is handled by the payload content, not the transport mechanism.

### Important Note on Missing Context
1.  **Refresh Token Table:** To properly implement rotation, it is best practice to have a separate `refresh_tokens` table. Using the `user` entity directly is fine for simple implementations, but it prevents the user from being logged into multiple devices simultaneously.
2.  **Auth Service:** The implementation of the `AuthServiceClient` (the actual controller/service handling the gRPC calls) was not provided. The logic for incrementing the `refresh_token_version` must be placed in the method that validates the refresh token.

### Security Recommendation
If you detect `decoded.version !== user.refresh_token_version`, you should **invalidate all active sessions** for that user (set `refresh_token_version` to 0 or a new random value) as this is a high-confidence indicator of a leaked token.
